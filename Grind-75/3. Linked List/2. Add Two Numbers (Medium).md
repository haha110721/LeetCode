```python
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:

        dummy = ListNode(0)
        cur = dummy
        
        carry = 0

        while l1 != None or l2 != None:
            if l1 != None:
                carry += l1.val
                l1 = l1.next
            if l2 != None:
                carry += l2.val
                l2 = l2.next
            
            cur.next = ListNode(carry % 10)
            carry = 1 if carry >= 10 else 0
            cur = cur.next

            if carry == 1:
                cur.next = ListNode(1)
            
        return dummy.next
```
