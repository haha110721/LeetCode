Number of Islands (Medium)
===

Problem: https://leetcode.com/problems/number-of-islands/description/

---

1. DFS
```python
# time: O(m × n)
    # 每個格子（cell）最多只會被拜訪一次
# space: O(m × n)
    # 每一次遞迴呼叫，都會 在系統堆疊（call stack）裡佔據一個空間，系統會幫你「暫存」這個遞迴函式的變數、參數、回傳點，直到遞迴結束才會釋放
    # 當你一口氣深入很多層（比如一條細長的島），堆疊空間就會不斷增加，直到最深的那一層
    # 空間複雜度 = 同時存在的最大遞迴深度

class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0

        row = len(grid)
        col = len(grid[0])

        def dfs(r, c):
            # 超出邊界 or 遇到 "0"，跳出
            if r < 0 or r >= row or c < 0 or c >= col:
                return 
            if grid[r][c] == "0":
                return 

            grid[r][c] = "0"  # 淹掉這塊陸地
            # 繼續往四個方向淹
            dfs(r+1, c)
            dfs(r-1, c)
            dfs(r, c+1)
            dfs(r, c-1)

        count = 0
        for i in range(row):
            for j in range(col):
                if grid[i][j] == "1":
                    count += 1  # 發現新島
                    dfs(i, j)  # 淹掉整座島
                    
        return count
```

```python
class Solution:
    def dfs(self, grid, i, j): # 找到 '1' 之後，擴散，把附近的 '1' 都變成 '0'，然後就會跳出這個 function
        grid[i][j] = 0
        for (dr, dc) in (1, 0), (-1, 0), (0, 1), (0, -1):
            row = i + dr
            col = j + dc
            if 0 <= row < len(grid) and 0 <= col < len(grid[0]) and grid[row][col] == '1':
                self.dfs(grid, row, col)

    def numIslands(self, grid: List[List[str]]) -> int:
        count = 0

        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == '1':
                    self.dfs(grid, i, j) # 每個格子都做，跳出 function 之後，就算找到一個 island，所以 count + 1
                    count += 1
        
        return count
```        

2. BFS (使用 queue)
```python
# time: O(m × n), space: O(m × n)

from collections import deque


class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0

        row = len(grid)
        col = len(grid[0])

        count = 0

        for i in range(row):
            for j in range(col):
                if grid[i][j] == "1":
                    count += 1  # 發現一座新島
                    queue = deque()
                    queue.append((i, j))
                    grid[i][j] = "0"

                    while queue:
                        r, c = queue.popleft()
                        
                        # 檢查四個方向
                        directions = [(1,0), (-1,0), (0,1), (0,-1)]
                        for dr, dc in directions:
                            nr, nc = r + dr, c + dc
                            if 0 <= nr < row and 0 <= nc < col and grid[nr][nc] == '1':
                                grid[nr][nc] = '0'  # 標記已拜訪
                                queue.append((nr, nc))

        return count
```
